            // the setup function runs once when you press reset or power the board
void setup() {
  
  pinMode(13, OUTPUT); // initializing pin
  pinMode(12, OUTPUT); // initializing pin
  pinMode(11, OUTPUT); // initializing pin
  
}

// the loop function runs over and over again forever
void loop() {
  digitalWrite(13, HIGH);  // turning on light
  delay(500); // wait .5 seconds
  digitalWrite(13, LOW); // turning off light
  delay(500); // wait .5 seconds
  digitalWrite(12, HIGH); //turn on light
  delay(500); // wait .5 seconds
  digitalWrite(12, LOW); // turn off light
  delay(500); // wait .5 seconds
  digitalWrite(11, HIGH); // turn on light
  delay(500); // wait .5 seconds
  digitalWrite(11, LOW);  //turn off light
  delay(500); // wait .5 seconds
